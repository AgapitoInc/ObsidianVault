    .map(f => f.basename);

if (!app.vault.getAbstractFileByPath(folderPath)) {
    await app.vault.createFolder(folderPath);
}

// Rename and move the file
await tp.file.rename(fileName);
await tp.file.move(`${folderPath}/${fileName}`);
